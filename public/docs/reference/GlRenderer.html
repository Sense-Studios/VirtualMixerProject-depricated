Example
<!-- a Canvas element with id: glcanvas is required! -->
   <canvas id="glcanvas"></canvas>


   <script>
     let renderer = new GlRenderer();

     var red = new SolidSource( renderer, { color: { r: 1.0, g: 0.0, b: 0.0 } } );
     let output = new Output( renderer, red )

     renderer.init();
     renderer.render();
   </script>
